import time
from selenium import webdriver
from selenium.webdriver.common.by import By

def main():
    """
    Main function to run the browser automation.
    """
    try:
        # --- CHANGE THIS LINE ---
        # Instead of webdriver.Chrome(), use webdriver.Edge()
        print("Attempting to start Microsoft Edge...")
        driver = webdriver.Edge()
        print("SUCCESS: Microsoft Edge driver started successfully!")
        
        # Now you can use the driver object just like before
        driver.get("https://www.bing.com") # Let's go to bing for fun
        print(f"Current page title: {driver.title}")
        time.sleep(5)

    except Exception as e:
        print(f"An error occurred: {e}")

    finally:
        if 'driver' in locals() and driver:
            print("Closing the browser.")
            driver.quit()

if __name__ == "__main__":
    main()
